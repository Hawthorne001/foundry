45fn resolve_rpc_url(chain: Chain, state: &mut crate::Cheatcodes) -> Result {
46    if let Some(config) = state.config.forks.get(&chain) {
47        let rpc = match config.rpc_endpoint {
48            Some(ref url) => url.clone().resolve(),
49            None => state.config.rpc_endpoint(&get_chain_name(&chain))?,
50        };
51
52        return Ok(rpc.url()?.abi_encode());
53    }
54
55    bail!(
56        "'rpc_endpoint' not found in '[fork.<chain_id: {chain}>]' subsection of 'foundry.toml'",
57        chain = chain.id()
58    )
59}

74/// Gets the alloy chain name for a given chain id.
75fn get_chain_name(chain: &Chain) -> Cow<'static, str> {
76    chain.named().map_or(Cow::Owned(chain.id().to_string()), |name| Cow::Borrowed(name.as_str()))
77}

79/// Gets the chain id of the active fork. Panics if no fork is selected.
80fn get_active_fork_chain_id(ccx: &mut CheatsCtxt) -> Result<u64> {
81    let (db, _, env) = ccx.as_db_env_and_journal();
82    if !db.is_forked_mode() {
83        bail!("a fork must be selected");
84    }
85    Ok(env.cfg.chain_id)
86}
87
88/// Gets the alloy chain for the active fork. Panics if no fork is selected.
89fn get_active_fork_chain(ccx: &mut CheatsCtxt) -> Result<Chain> {
90    get_active_fork_chain_id(ccx).map(Chain::from_id)
91}

95    ($struct:ident, $sol_type:expr,stateful) => {
96        impl Cheatcode for $struct {
97            fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
98                let Self { key } = self;
99                let chain = get_active_fork_chain(ccx)?;
100                get_value(chain, key, $sol_type, ccx.state)
101            }
102        }
103    };
104    ($struct:ident, $sol_type:expr) => {
105        impl Cheatcode for $struct {
106            fn apply(&self, state: &mut crate::Cheatcodes) -> Result {
107                let Self { chain, key } = self;
108                get_value(Chain::from_id(chain.to::<u64>()), key, $sol_type, state)
109            }
110        }
111    };
112}

175/// Generic helper to get any value from the TOML config.
176fn get_toml_value<'a>(
177    chain: Chain,
178    key: &'a str,
179    state: &'a crate::Cheatcodes,
180) -> Result<&'a toml::Value> {
181    let config = state.config.forks.get(&chain).ok_or_else(|| {
182        fmt_err!(
183            "'[fork.<chain_id: {chain}>]' subsection not found in 'foundry.toml'",
184            chain = chain.id()
185        )
186    })?;
187    let value = config.vars.get(key).ok_or_else(|| {
188        fmt_err!("variable '{key}' not found in '[fork.<chain_id: {chain}>]'", chain = chain.id())
189    })?;
190
191    Ok(value)
192}

194/// Generic helper to get any single value from the TOML config.
195fn get_value(chain: Chain, key: &str, ty: &DynSolType, state: &crate::Cheatcodes) -> Result {
196    let value = get_toml_value(chain, key, state)?;
197    let sol_value = parse_toml_element(value, ty, key, chain)?;
198    Ok(sol_value.abi_encode())
199}

201/// Generic helper to get an array of values from the TOML config.
202fn get_array(
203    chain: Chain,
204    key: &str,
205    element_ty: &DynSolType,
206    state: &crate::Cheatcodes,
207) -> Result {
208    let arr = get_toml_value(chain, key, state)?.as_array().ok_or_else(|| {
209        fmt_err!(
210            "variable '{key}' in '[fork.<chain_id: {chain}>]' must be an array",
211            chain = chain.id()
212        )
213    })?;
214    let result: Result<Vec<_>> = arr
215        .iter()
216        .enumerate()
217        .map(|(i, elem)| {
218            let context = format!("{key}[{i}]");
219            parse_toml_element(elem, element_ty, &context, chain)
220        })
221        .collect();
222
223    Ok(DynSolValue::Array(result?).abi_encode())
224}

226/// Parses a single TOML value into a specific Solidity type.
227fn parse_toml_element(
228    elem: &toml::Value,
229    element_ty: &DynSolType,
230    key: &str,
231    chain: Chain,
232) -> Result<DynSolValue> {
233    // Convert TOML value to JSON value and use existing JSON parsing logic
234    parse_json_as(&toml_to_json_value(elem.to_owned()), element_ty).map_err(|e| {
235        fmt_err!("failed to parse '{key}' in '[fork.<chain_id: {chain}>]': {e}", chain = chain.id())
236    })
